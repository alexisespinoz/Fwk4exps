from fwk4exps.SpeculativeMonitor import *

import copy

def experimentalDesign():
   	
    params = {"__a" : 0.0, "__b" : 0.0, "__g" : 0.0, "__p" : 0.0}

    print "El mejor valor de p es: " + str(Sbest.params["__p"])
